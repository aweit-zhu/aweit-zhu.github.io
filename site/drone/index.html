3.透過 Docker Compose 建立 drone server

  • 開啟 CMD，輸入 WSL。

  • 切到 home 路徑。

    cd ~

  • 建立 drone-server 資料夾

    mkdir drone-server
cd drone-server

  • 建立 docker-compose.github.yml

    nano docker-compose.github.yml

貼上以下

version: '2'
services:
drone-server:
    /assets/image: drone/drone:1
    ports:
    - 8082:80
    volumes:
    - ./:/data
    restart: always
    environment:
    - DRONE_SERVER_HOST=${DRONE_SERVER_HOST}
    - DRONE_SERVER_PROTO=${DRONE_SERVER_PROTO}
    - DRONE_RPC_SECRET=${DRONE_RPC_SECRET}
    - DRONE_GITHUB_SERVER=<https://github.com>
    - DRONE_GITHUB_CLIENT_ID=${DRONE_GITHUB_CLIENT_ID}
    - DRONE_GITHUB_CLIENT_SECRET=${DRONE_GITHUB_CLIENT_SECRET}
    - DRONE_LOGS_PRETTY=true
    - DRONE_LOGS_COLOR=true
    - DRONE_USER_CREATE=username:aweit-zhu,admin:true

runner for docker version

drone-runner:
    /assets/image: drone/drone-runner-docker:1
    restart: always
    depends_on:
    - drone-server
    volumes:
    - /var/run/docker.sock:/var/run/docker.sock
    environment:
    - DRONE_RPC_HOST=${DRONE_RPC_HOST}
    - DRONE_RPC_PROTO=${DRONE_RPC_PROTO}
    - DRONE_RPC_SECRET=${DRONE_RPC_SECRET}
    - DRONE_RUNNER_CAPACITY=3

4. 撰寫 .drone.yml 檔

  • 建立 ssh_password_master secret

  • 建立 ssh_password secret

  • 解析 .drone.yml (略)

    kind: pipeline
type: docker
name: docker-demo-drone
steps:

- name: test

        path: /app/build
    commands:
- mvn install -DskipTests=true -Dmaven.javadoc.skip=true -B -V

- name: package

- name: maven-cache

- cp run.sh /app/build/run.sh

- name: scp files
